<script>
  // Script de redirección para manejar slugs traducidos indexados por error
  (function() {
    const path = window.location.pathname;
    
    // Mapeo de fragmentos de slugs traducidos a sus versiones originales en español
    // Esto ayuda a redirigir URLs como /de/blog/DeepSeek-Detaillierte_Analyse... 
    // a /de/blog/DeepSeek-Análisis_detallado...
    const redirects = {
      'DeepSeek-Detaillierte_Analyse': 'DeepSeek-An%C3%A1lisis_detallado_del_Proyecto_.IA_Isla_Anticitera',
      'Detailed_Analysis': 'An%C3%A1lisis_detallado_del_Proyecto_.IA_Isla_Anticitera',
      'Analyse_detaillee': 'An%C3%A1lisis_detallado_del_Proyecto_.IA_Isla_Anticitera',
      'Analisi_dettagliata': 'An%C3%A1lisis_detallado_del_Proyecto_.IA_Isla_Anticitera'
    };

    for (const [key, value] of Object.entries(redirects)) {
      if (path.includes(key)) {
        // Mantener el prefijo de idioma y corregir el slug
        const segments = path.split('/');
        // Buscar el segmento que contiene la clave y reemplazarlo
        const newSegments = segments.map(s => s.includes(key) ? value : s);
        const newPath = newSegments.join('/');
        
        console.log('Redirecting to:', newPath);
        window.location.replace(newPath);
        return;
      }
    }
  })();
</script>
